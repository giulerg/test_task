  from items
  WHERE update_date >= '2020-06-01'
  group by item_id) as actual 
on items.item_id = actual.item_id 
and items.update_date = actual.update_date

  from items
  WHERE price >= 3
  group by item_id
) as actual 
on items.item_id = actual.item_id 
and items.update_date = actual.update_date
